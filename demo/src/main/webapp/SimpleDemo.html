Here is the used grammar:

 root : expr+;

 expr : 
	cell
 |	NUMBER
 |	STRING
 |	WORD
 ;

 cell : LEFT_P first_cell params+=expr (params+=expr)* RIGHT_P;

 first_cell : 
 	cell
 |	OP
 |	WORD
 ;

 LEFT_P :  '(';
 RIGHT_P:  ')';
 OP     :  '+' | '-' | '*' | '/' | 'mod';
 STRING :  '\''~[']*'\'';
 WORD   :  [a-zA-Z0-9_]+;
 NUMBER :  [0-9]+;
 WS     :  [ \t\n\r]+	->	skip;
